# 1 字符串分割
没有空格    
```shell
#!/bin/bash
var=’1,2,3,4,5’
var=${var//,/ }    # 将 var 中的,替换为空格
for element in $var
do
    echo $element
done
```

有空格    

```shell
echo 'mark:x:0:0:this is a test user:/var/mark:nologin' |cut -d ":" -f i # 输出第 i 个字符
```

```shell
#!/bin/bash
user=’mark:x:0:0:this is a test user:/var/mark:nologin’
i=1
while((1==1))
do
        split=`echo $user|cut -d ":" -f$i`
        if [ "$split" != "" ]
        then
                ((i++))
                echo $split
        else
                break
        fi
done
```
